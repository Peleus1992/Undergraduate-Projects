00019 // A union that describes the entries of the vector table.  The union is needed
00020 // since the first entry is the stack pointer and the remainder are function
00021 // pointers.
00022 //
00023 // Normally the vectorTable below would require entries such as:
00024 //   { .topOfStack = x },
00025 //   { .ptrToHandler = y },
00026 // But since ptrToHandler is defined first in the union, this is the default
00027 // type which means we don't need to use the full, explicit entry.  This makes
00028 // the vector table easier to read because it's simply a list of the handler
00029 // functions.  topOfStack, though, is the second definition in the union so
00030 // the full entry must be used in the vectorTable.
00031 //=============================================================================
00032 typedef union
00033 {
00034   void (*ptrToHandler)(void);
00035   void *topOfStack;
00036 } HalVectorTableType;
